HDU_1166 敌兵布阵 【线段树】
<!--more-->

>[原题链接](http://acm.hdu.edu.cn/showproblem.php?pid=1166)

### 题目意思： ###
给定一个序列，有三种操作，最后输出给定范围的值之和
### 解题思路： ###
数据大，普通方法肯定超时，用线段树来计算区间的加法。
### 代码部分： ###

```cpp
#include <iostream>
#include <stdio.h>
#include <string.h>
#define lchild left,mid,root<<1   //宏定义代替结构体
#define rchild mid+1,right,root<<1|1
using namespace std;

const int maxn=50005;
int Sum[maxn<<2];
int n;
//更新线段树节点
void push_up(int root)
{
    Sum[root] = Sum[root<<1]+Sum[root<<1|1];
    return ;
}
//构建线段树
void build(int left,int right ,int root)
{
    if(left == right)
    {
        int num;
        scanf("%d",&num);
        Sum[root] = num;
        return ;
    }
    int mid = (left+right)>>1;
    build(lchild);
    build(rchild);
    push_up(root);
}

//更改操作
void update(int aim,int num,int left,int right,int root)
{
    if(right == left)
    {
        Sum[root] += num;
        return ;
    }
    int mid = (left+right)>>1;
    if(mid>=aim)
        update(aim,num,lchild);
    else
        update(aim,num,rchild);
    push_up(root);
}
//查询操作
int query(int L,int R,int left,int right,int root)
{
    if(L <= left&&right <= R)
    {
        return Sum[root];
    }
    int mid = (left+right)>>1;
    int ans = 0;
    //ans+=(L,R,lchild);
    //ans+=(L,R,rchild);
    if(R <= mid)
    {
        ans += query(L,R,lchild);
    }
    else if(L > mid)
    {
        ans += query(L,R,rchild);
    }
    else
    {
        ans += query(L,R,lchild);
        ans += query(L,R,rchild);
    }
    return ans;
}
int main ()
{
    int t,temp=1;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        build(1,n,1);
        string s;
        cout<<"Case "<<temp++<<":"<<endl;
        while(cin>>s)
        {
            int n1,n2;
            if(s[0] == 'E')
                break;
            if(s[0] == 'Q')
            {
                scanf("%d %d",&n1,&n2);
                int ans = query(n1,n2,1,n,1);
                cout<<ans<<endl;
            }
            if(s[0] == 'A')
            {
                scanf("%d %d",&n1,&n2);
                update(n1,n2,1,n,1);
            }
            if(s[0] == 'S')
            {
                scanf("%d %d",&n1,&n2);
                update(n1,-n2,1,n,1);
            }
        }
    }

    return 0;
}
```
